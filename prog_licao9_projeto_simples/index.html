2 Função para Cadastrar Estudante

Criamos uma função que recebe nome e notas, calcula a média automaticamente e adiciona ao array.

JavaScript 
function cadastrarEstudante(nome, notas) {
    // Calcular média usando reduce
    let soma = notas.reduce((acc, nota) => acc + nota, 0);
    let media = soma / notas.length;
    
    // Determinar situação (condicional)
    let situacao = "";
    if (media >= 7) {
        situacao = "Aprovado";
    } else if (media >= 5) {
        situacao = "Recuperação";
    } else {
        situacao = "Reprovado";
    }
    
    // Criar objeto e adicionar ao array
    let estudante = {
        nome: nome,
        notas: notas,
        media: media.toFixed(1),
        situacao: situacao
    };
    
    estudantes.push(estudante);
    console.log("✅ Estudante cadastrado com sucesso!");
}

3 Função para Listar Estudantes (Loop)

Usamos um loop para percorrer o array e exibir todos os estudantes cadastrados.

JavaScript 
function listarEstudantes() {
    if (estudantes.length === 0) {
        console.log("Nenhum estudante cadastrado.");
        return;
    }
    
    console.log("\\n📋 LISTA DE ESTUDANTES:");
    console.log("========================");
    
    // Loop for...of para percorrer array
    for (let i = 0; i < estudantes.length; i++) {
        let e = estudantes[i];
        console.log("${i + 1}. ${e.nome}");
        console.log("   Média: ${e.media} - ${e.situacao}");
        console.log("");
    }
}

4 Função de Busca (Filter)

Criamos uma função para buscar estudantes por nome usando o método filter.

JavaScript 
function buscarEstudante(nomeBusca) {
    // Filter retorna array com elementos que atendem condição
    let encontrados = estudantes.filter(e => 
        e.nome.toLowerCase().includes(nomeBusca.toLowerCase())
    );
    
    if (encontrados.length > 0) {
        console.log("🔍 Resultados encontrados:");
        encontrados.forEach(e => {
            console.log("- ${e.nome}: ${e.situacao}");
        });
    } else {
        console.log("❌ Nenhum estudante encontrado.");
    }
    
    return encontrados;
}

5 Estatísticas do Sistema (Reduce e Math)

Função que calcula estatísticas gerais: média da turma, maior nota, menor nota, total de aprovados.

JavaScript 
function gerarEstatisticas() {
    if (estudantes.length === 0) {
        console.log("Sem dados para estatísticas.");
        return;
    }
    
    // Extrair todas as médias
    let medias = estudantes.map(e => parseFloat(e.media));
    
    // Cálculos
    let mediaTurma = medias.reduce((a, b) => a + b, 0) / medias.length;
    let maiorMedia = Math.max(...medias);
    let menorMedia = Math.min(...medias);
    let aprovados = estudantes.filter(e => e.situacao === "Aprovado").length;
    
    console.log("\\n📊 ESTATÍSTICAS DA TURMA:");
    console.log("Total de estudantes: ${estudantes.length}");
    console.log("Média da turma: ${mediaTurma.toFixed(1)}");
    console.log("Maior média: ${maiorMedia}");
    console.log("Menor média: ${menorMedia}");
    console.log("Aprovados: ${aprovados} (${((aprovados/estudantes.length)*100).toFixed(0)}%)");
}

Código Completo

Aqui está o sistema completo, pronto para usar:

JavaScript - Sistema Completo 
// ==========================================
// SISTEMA DE GERENCIAMENTO DE ESTUDANTES
// ==========================================

let estudantes = [];

// 1. CADASTRAR ESTUDANTE
function cadastrarEstudante(nome, notas) {
    let soma = notas.reduce((acc, nota) => acc + nota, 0);
    let media = soma / notas.length;
    
    let situacao = media >= 7 ? "Aprovado" : 
                 media >= 5 ? "Recuperação" : "Reprovado";
    
    estudantes.push({
        nome: nome,
        notas: notas,
        media: media.toFixed(1),
        situacao: situacao
    });
    
    console.log("✅ ${nome} cadastrado!");
}

// 2. LISTAR TODOS
function listarEstudantes() {
    console.log("\\n📋 LISTA DE ESTUDANTES:");
    estudantes.forEach((e, i) => {
        console.log("${i + 1}. ${e.nome} - Média: ${e.media} (${e.situacao})");
    });
}

// 3. BUSCAR ESTUDANTE
function buscarEstudante(termo) {
    let resultados = estudantes.filter(e => 
        e.nome.toLowerCase().includes(termo.toLowerCase())
    );
    return resultados;
}

// 4. ESTATÍSTICAS
function estatisticas() {
    if (estudantes.length === 0) return;
    
    let medias = estudantes.map(e => parseFloat(e.media));
    let mediaTurma = medias.reduce((a, b) => a + b) / medias.length;
    let aprovados = estudantes.filter(e => e.situacao === "Aprovado").length;
    
    console.log("\\n📊 Estatísticas:");
    console.log("Média da turma: ${mediaTurma.toFixed(1)}");
    console.log("Taxa de aprovação: ${((aprovados/estudantes.length)*100).toFixed(0)}%");
}

// ===== TESTANDO O SISTEMA =====

cadastrarEstudante("Ana Silva", [8.5, 9.0, 7.5]);
cadastrarEstudante("Bruno Costa", [6.0, 5.5, 7.0]);
cadastrarEstudante("Carla Mendes", [9.5, 9.0, 10]);
cadastrarEstudante("Daniel Souza", [4.0, 5.0, 3.5]);

listarEstudantes();
estatisticas();
